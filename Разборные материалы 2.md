#include <vector>
#include <iostream>
#include <bitset>
using namespace std;

class Bloom
{
	vector<bool> __bloom;
	int hash_foo(int a, int b, string key)
	{
		int Hash = 0;
		for (size_t i = 0; i < key.size(); i++)
		{
			Hash += static_cast<int>(key[i]);
		}
		return (a * Hash + b) % 50;
	}
public:
	Bloom() = default;
	Bloom(size_t size)
	{
		__bloom.resize(size, false) 
	}
	void insert_key(string key)
	{
		__bloom[hash_foo(11, 13, key)] = true;
		__bloom[hash_foo(17, 19, key)] = true;
		__bloom[hash_foo(23, 29, key)] = true;
		__bloom[hash_foo(31, 37, key)] = true;
		__bloom[hash_foo(41, 43, key)] = true;
	}
	
	bool find_key(string key)
	{
		bool result = true;
		result = result && __bloom[hash_foo(11, 13, key)]
			&& __bloom[hash_foo(17, 19, key)]
			&& __bloom[hash_foo(23, 29, key)]
			&& __bloom[hash_foo(31, 37, key)]
			&& __bloom[hash_foo(41, 43, key)];
		return result;
	}
};

{
